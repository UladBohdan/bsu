\section{Яўны метад Эйлера}
Ураўненне:
\begin{equation}
    y_{n+1} = y_n + hf_n, \hspace{10pt} h = \frac{x_l-x_0}{N}, \hspace{10pt} n = \overline{0, N-1}. \\
\end{equation}

\begin{equation}
	\begin{cases}
y_{n+1} = y_n + hf_n, \\
z_{n+1} = z_n + hg_n, \hspace{10pt} h = \frac{x_l-x_0}{N}, \hspace{10pt} n = \overline{0, N-1}.
	\end{cases}
\end{equation}
